AngularJS filter 使用常用方法

not equal

html

<div ng-app>
    <div ng-controller="TodoCtrl">
         <h1>List</h1>
        <div ng-repeat="t in todos | filter:{ id: '!-1'}"></div>
    </div>
</div>

js

function TodoCtrl($scope) {
    $scope.todos = [{
        text: 'learn angular',
        done: true,
        id: -1
    }, {
        text: 'build an angular app',
        done: false,
        id: 1
    }];
}
