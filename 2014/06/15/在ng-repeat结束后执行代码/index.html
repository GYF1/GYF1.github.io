在ng-repeat结束后执行代码

很多时候发现有些功能在写死的数据下可以，但是一旦使用了ng-repeat以后就不行了。
很常见的就是jquery的一些lib。
问题出在执行的顺序，jquery的初始化功能的代码在ng-repeat完成之前。
所以我们需要控制执行的顺序，在ng-repeat完成后，在做相关的代码

javascript

var module = angular.module('testApp', [])
    .directive('onFinishRender', function ($timeout) {
    return {
        restrict: 'A',
        link: function (scope, element, attr) {
            if (scope.$last === true) {
                $timeout(function () {
                    scope.$emit(attr.onFinishRender);
                });
            }
        }
    }
});

function myC($scope) {
    $scope.ta = [1, 2, 3, 4, 5, 6];
    $scope.$on('test', function(ngRepeatFinishedEvent) {
        console.log($("p"));
        console.log("done");
    });
}

html

<div ng-app="testApp" ng-controller="myC">
    <p ng-repeat="t in ta" on-finish-render="test"></p>
</div>
