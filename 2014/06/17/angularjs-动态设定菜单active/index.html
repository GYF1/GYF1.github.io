AngularJS 动态设定菜单active

根据当前的url，动态设定菜单项的激活状态

html

ng-class="{active:isActive('/dashboard')}"

js

myApp.controller('MyCtrl', function($scope, $location) {
    $scope.isActive = function(route) {
        return route === $location.path();
    }
});
